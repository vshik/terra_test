# -------------------------------------------------------------
# LLM helper â€“ enforce strict JSON mode
# -------------------------------------------------------------
async def send_to_llm(messages, expect_json=False):
    if expect_json:
        response = await llm_client.chat.completions.create(
            model=MODEL_NAME,
            messages=messages,
            response_format={"type": "json_object"}  # âœ… Hard-enforce JSON mode
        )
    else:
        response = await llm_client.chat.completions.create(
            model=MODEL_NAME,
            messages=messages
        )
    return response.choices[0].message.content.strip()


# -------------------------------------------------------------
# JSON normalization â€“ fix synonyms and variants
# -------------------------------------------------------------
def normalize_tool_json(raw: str) -> dict:
    """Force output to use {tool, params} keys only."""
    cleaned = raw.strip().replace("'", '"')

    try:
        data = json.loads(cleaned)
    except json.JSONDecodeError:
        return {"tool": "none"}

    # ðŸ”§ Normalize synonyms
    if "action" in data and "tool" not in data:
        data["tool"] = data.pop("action")

    if "parameters" in data and "params" not in data:
        data["params"] = data.pop("parameters")

    # ðŸ”§ Handle missing params key
    if "params" not in data:
        data["params"] = {}

    return data
